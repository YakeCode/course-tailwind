***     CONFIGURAR  VARIABLES

    vamos al archivo de configuracion 'tailwind.config.js'

    dentro de /'theme : { extend:{} }'/ ingresamos las variables en extend

*** imagenes de fondo

    backgroundImage: {
        
        'nombreDeLaImagen':"url( ubicacion del archivo)",   con la coma separamos las imagenes

        ** 'sanFrancisco': "url('../img/sanFrancisco.jpg')", **
    },

*** Colores de fondo

    sintaxis    backgroundColor: theme =>({
                    ...theme('colors'),
                    'nombreDelColor': 'color',
                }),

    backgroundColor: theme => ({
      ...theme('colors'),
      'primary': '#cc2d4a',
    }),

*** Color de texto

    *** textColor:{ 'nombreDelColor': 'Color'} ***
    
    textColor:{
      'primary': '#cc2d4a',
    },
